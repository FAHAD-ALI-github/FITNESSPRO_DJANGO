# NEW MODEL: Trainer Attendance
class TrainerAttendance(models.Model):
    trainer = models.ForeignKey(Trainer, on_delete=models.CASCADE)
    date = models.DateField()
    check_in_time = models.TimeField(null=True, blank=True)
    check_out_time = models.TimeField(null=True, blank=True)
    is_present = models.BooleanField(default=False)
    notes = models.TextField(blank=True)
    
    class Meta:
        unique_together = ['trainer', 'date']
        ordering = ['-date']
    
    def __str__(self):
        return f"{self.trainer.first_name} - {self.date} - {'Present' if self.is_present else 'Absent'}"
    
    @staticmethod
    def get_attendance_by_trainer(trainer_id, start_date=None, end_date=None):
        query = TrainerAttendance.objects.filter(trainer_id=trainer_id)
        if start_date:
            query = query.filter(date__gte=start_date)
        if end_date:
            query = query.filter(date__lte=end_date)
        return query
    
    @staticmethod
    def mark_attendance(trainer_id, date, check_in=None, check_out=None):
        attendance, created = TrainerAttendance.objects.get_or_create(
            trainer_id=trainer_id,
            date=date,
            defaults={'is_present': True, 'check_in_time': check_in}
        )
        if not created and check_out:
            attendance.check_out_time = check_out
            attendance.save()
        return attendance

def calculate_bmi(weight, height):
    """Calculate BMI and return BMI value and category"""
    bmi = weight / ((height / 100) ** 2)
    bmi = round(bmi, 2)
    
    if bmi < 18.5:
        category = "Underweight"
    elif bmi < 25:
        category = "Normal weight"
    elif bmi < 30:
        category = "Overweight"
    else:
        category = "Obese"
    
    return bmi, category


def calculate_membership_dates(start_date, months):
    """Calculate membership end date based on start date and duration"""
    end_date = start_date + timedelta(days=30 * months)
    return end_date

def pre_register_view(request):
    """Step 1: User selects membership plan and enters basic info"""
    data = {}
    
    if request.method == "POST":
        # Get form data
        first_name = request.POST.get("firstname")
        last_name = request.POST.get("lastname")
        email = request.POST.get("email")
        phone = request.POST.get("phone")
        plan_id = request.POST.get("membership_plan")
        
        # Validation
        error = ""
        if not all([first_name, last_name, email, phone, plan_id]):
            error = "All fields are required"
        elif len(first_name) < 2 or len(last_name) < 2:
            error = "Name must be at least 2 characters"
        elif len(phone) < 10:
            error = "Invalid phone number"
        
        if error:
            data["error"] = error
            data["first_name"] = first_name
            data["last_name"] = last_name
            data["email"] = email
            data["phone"] = phone
        else:
            # Store in session for next step
            request.session['prereg_data'] = {
                'first_name': first_name.strip(),
                'last_name': last_name.strip(),
                'email': email.strip(),
                'phone': phone.strip(),
                'plan_id': plan_id
            }
            return redirect('/payment-upload')
    
    # Get active membership plans
    data['plans'] = MembershipPlan.get_active_plans()
    return render(request, 'pre_register.html', data)


def payment_proof_upload(request):
    """Step 2: User uploads payment proof"""
    data = {}
    
    # Check if pre-registration data exists
    prereg_data = request.session.get('prereg_data')
    if not prereg_data:
        return redirect('/pre-register')
    
    if request.method == "POST":
        payment_image = request.FILES.get('payment_image')
        amount_paid = request.POST.get('amount_paid')
        
        error = ""
        if not payment_image:
            error = "Payment proof image is required"
        elif not amount_paid:
            error = "Amount paid is required"
        
        if error:
            data["error"] = error
        else:
            try:
                # Create PaymentProof record
                plan = MembershipPlan.objects.get(id=prereg_data['plan_id'])
                
                payment_proof = PaymentProof.objects.create(
                    temp_first_name=prereg_data['first_name'],
                    temp_last_name=prereg_data['last_name'],
                    temp_email=prereg_data['email'],
                    temp_phone=prereg_data['phone'],
                    membership_plan=plan,
                    payment_image=payment_image,
                    amount_paid=amount_paid,
                    status='pending'
                )
                
                # Clear session data
                del request.session['prereg_data']
                
                # Store payment proof ID for status checking
                request.session['payment_proof_id'] = payment_proof.id
                
                return redirect('/registration-pending')
                
            except Exception as e:
                data["error"] = f"Error: {str(e)}"
    
    data['prereg_data'] = prereg_data
    data['plan'] = MembershipPlan.objects.get(id=prereg_data['plan_id'])
    return render(request, 'payment_upload.html', data)

def login(request):
    """User login - Modified to check approval status"""
    data = {}
    
    if request.method == "POST":
        username = request.POST.get("u_username")
        password = request.POST.get("u_password")
        
        user_data = Gym_user.get_user_by_username(username)
        
        if user_data:
            # Check if user is approved
            if not user_data.is_approved:
                data["error"] = "Your account is pending admin approval. Please check your registration status."
                return render(request, 'login_page.html', data)
            
            if user_data.password == password:
                # Set session
                request.session['user_id'] = user_data.id
                request.session['user_username'] = user_data.username
                
                # Calculate BMI
                bmi, category = calculate_bmi(user_data.weight, user_data.height)
                
                # Get current day
                today = datetime.now()
                
                data['bmi'] = bmi
                data['bmi_status'] = category
                data['today'] = today.strftime("%A")
                data['user'] = user_data
                
                return render(request, "profile_page.html", data)
            else:
                data["error"] = "Password is incorrect!"
        else:
            data["error"] = "Incorrect username or password!"
    
    return render(request, 'login_page.html', data)


def user_logout(request):

    if 'user_username' in request.session:

    return redirect('/')

@csrf_exempt
@require_POST
def upload_profile_image(request):
    """Upload user profile image"""
    try:
        user_id = request.POST.get('user_id') or request.GET.get('u_id')
        
        if not user_id:
            return JsonResponse({'success': False, 'message': 'User ID is required.'})
        
        user = Gym_user.get_user_by_id(int(user_id))
        if not user:
            return JsonResponse({'success': False, 'message': 'User does not exist.'})
        
        if 'image' in request.FILES:
            image_file = request.FILES['image']
            
            # Validate file type
            allowed_types = ['image/jpeg', 'image/jpg', 'image/png', 'image/gif', 'image/webp']
            if image_file.content_type not in allowed_types:
                return JsonResponse({'success': False, 'message': 'Invalid file type.'})
            
            # Validate file size (5MB limit)
            if image_file.size > 5 * 1024 * 1024:
                return JsonResponse({'success': False, 'message': 'File too large. Max 5MB.'})
            
            # Delete old image if not default
            if user.image and user.image.name != "users_profile_images/image.png":
                try:
                    user.image.delete(save=False)
                except:
                    pass
            
            # Save new image
            user.image.save(image_file.name, image_file, save=True)
            
            return JsonResponse({
                'success': True,
                'message': 'Profile picture updated successfully!',
                'image_url': user.image.url
            })
        else:
            return JsonResponse({'success': False, 'message': 'No image file provided.'})
            
    except ValueError:
        return JsonResponse({'success': False, 'message': 'Invalid user ID format.'})
    except Exception as e:
        return JsonResponse({'success': False, 'message': f'An error occurred: {str(e)}'})

@trainer_login_required
def edit_diet_plan(request, plan_id):
    """Edit diet plan"""
    plan = DietPlan.objects.get(id=plan_id)
    
    # Ensure trainer owns this plan
    if plan.trainer_id != request.session.get('trainer_id'):
        return redirect('/view-diet-plans')
    
    data = {'plan': plan}
    
    if request.method == "POST":
        plan.plan_name = request.POST.get("plan_name")
        plan.target_goal = request.POST.get("target_goal")
        plan.calorie_range = request.POST.get("calorie_range")
        plan.meal_breakdown = request.POST.get("meal_breakdown")
        plan.protein_grams = request.POST.get("protein")
        plan.carbs_grams = request.POST.get("carbs")
        plan.fats_grams = request.POST.get("fats")
        plan.notes = request.POST.get("notes")
        plan.save()
        
        return redirect('/view-diet-plans')
    
    return render(request, 'edit_diet_plan.html', data)


@trainer_login_required
def delete_diet_plan(request, plan_id):
    """Delete diet plan"""
    plan = DietPlan.objects.get(id=plan_id)
    
    # Ensure trainer owns this plan
    if plan.trainer_id == request.session.get('trainer_id'):
        plan.delete()
    
    return redirect('/view-diet-plans')

@trainer_login_required
def trainer_mark_attendance(request):
    """Trainer marks own attendance"""
    trainer_id = request.session.get('trainer_id')
    trainer = Trainer.objects.get(id=trainer_id)
    
    data = {'trainer': trainer}
    
    today = date.today()
    
    # Check if already marked today
    existing = TrainerAttendance.objects.filter(
        trainer_id=trainer_id,
        date=today
    ).first()
    
    if existing:
        data['already_marked'] = True
        data['attendance'] = existing
    
    if request.method == "POST":
        action = request.POST.get("action")
        
        if action == "check_in":
            from datetime import datetime as dt
            check_in_time = dt.now().time()
            
            if existing:
                existing.check_in_time = check_in_time
                existing.is_present = True
                existing.save()
            else:
                TrainerAttendance.objects.create(
                    trainer_id=trainer_id,
                    date=today,
                    check_in_time=check_in_time,
                    is_present=True
                )
            
            data['msg'] = "Check-in recorded successfully!"
            
        elif action == "check_out" and existing:
            from datetime import datetime as dt
            existing.check_out_time = dt.now().time()
            existing.save()
            data['msg'] = "Check-out recorded successfully!"
        
        # Refresh attendance data
        existing = TrainerAttendance.objects.filter(
            trainer_id=trainer_id,
            date=today
        ).first()
        data['attendance'] = existing
    
    # Get recent attendance history
    recent_attendance = TrainerAttendance.objects.filter(
        trainer_id=trainer_id
    ).order_by('-date')[:10]
    
    data['recent_attendance'] = recent_attendance
    
    return render(request, 'trainer_mark_attendance.html', data)
